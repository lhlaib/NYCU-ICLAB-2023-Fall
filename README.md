    def __init__(self, filename):
        self.f = open(filename, 'w')

    def write_case(self, inputs):
        # e.g., inputs = {'in_valid': 1, 'op': 2, 'data': 14}
        line = f"{inputs['in_valid']}_{inputs['op']}_{inputs['data']}\n"
        self.f.write(line)

    def close(self):
        self.f.close()
